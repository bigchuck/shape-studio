# Random Triangle Generator for LoRA Training
# Creates randomized triangles and promotes to MAIN

# Clear WIP before starting
CLEAR WIP ALL

# Create a random triangle
POLY tri RAND(100,668),RAND(100,668) RAND(100,668),RAND(100,668) RAND(100,668),RAND(100,668)

# Apply random transformations
ROTATE tri RAND(0,360)
SCALE tri RAND(0.7,1.3)

# Promote to MAIN canvas
PROMOTE tri

# Clean up WIP
CLEAR WIP ALL
