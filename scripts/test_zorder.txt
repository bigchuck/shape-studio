# Z-Order System Test Script

# ========================================
# TEST 1: Basic Auto-Assignment
# ========================================
CLEAR WIP ALL

# Create 3 shapes - should get z=0,1,2 automatically
POLY shape1 100,100 200,200 300,100
POLY shape2 150,150 250,250 350,150  
POLY shape3 200,200 300,300 400,200

# Verify auto-assignment
INFO shape1
INFO shape2  
INFO shape3
fill shape1 red
fill shape2 green
fill shape3 blue

# Visual check: shape1 should be furthest back, shape3 in front
SAVE test1_auto_assignment.png
# ========================================
# TEST 2: Manual Override (Counter Stays)
# ========================================
CLEAR main ALL

POLY auto1 100,100 200,200 300,100
# auto1 gets z=0

ZORDER auto1 100
# auto1 now has z=100, counter stays at 1

POLY auto2 150,150 250,250 350,150
# auto2 should get z=1 (counter didn't advance)

INFO auto1
INFO auto2
fill auto1 red
fill auto2 green

# Visual check: auto2 (z=1) should be behind auto1 (z=100)
SAVE test2_manual_override.png

# ========================================
# TEST 4: Member Z-Order Within Group
# ========================================
CLEAR main ALL

# Create members with explicit z-orders
POLY member_front 100,100 200,200 300,100
POLY member_middle 150,150 250,250 350,150
POLY member_back 200,200 300,300 400,200
fill member_front red
fill member_middle green
fill member_back blue

# Set explicit z-orders (reverse of creation order)
ZORDER member_front 10

ZORDER member_back 1

# Group them
GROUP layered member_front member_middle member_back

INFO layered
INFO member_front
INFO member_middle
INFO member_back

# Visual check: member_back (z=1) behind member_middle (z=5) behind member_front (z=10)
SAVE test4_member_zorder.png
# ========================================
# TEST 5: Extraction
# ========================================
CLEAR main ALL

POLY shape_a 100,100 200,200 300,100
POLY shape_b 400,400 500,500 600,400
GROUP mygroup shape_a shape_b

fill shape_a red
fill shape_b green

# Create shape between group's range
POLY shape_c 150,150 250,250 350,150
fill shape_c blue

# shape_c should get next z from counter
INFO mygroup
INFO shape_c

# Extract shape_a - it should keep its z-coord and participate in global sort
EXTRACT shape_a FROM mygroup

INFO shape_a
INFO mygroup

# Visual check: Extraction preserves z-coord
SAVE test5_extraction.png

# ========================================
# TEST 6: Complex Layering
# ========================================
CLEAR main ALL

# Background layer (explicit low z)
POLY bg 0,0 768,0 768,768 0,768
FILL bg lightgray
ZORDER bg -10

# Automatic middle layers
PROC dynamic_polygon mid1 VERTICES=5 BOUNDS=100,100,300,300
PROC dynamic_polygon mid2 VERTICES=6 BOUNDS=400,100,600,300
PROC dynamic_polygon mid3 VERTICES=7 BOUNDS=100,400,300,600

COLOR mid1 blue
COLOR mid2 green  
COLOR mid3 red

# Foreground layer (explicit high z)
POLY fg 200,200 568,200 568,568 200,568
# ALPHA fg 0.5
# ALPHA fg 0.
FILL fg yellow
ZORDER fg 4448

INFO bg
INFO mid1
INFO mid2
INFO mid3
INFO fg

# Visual check: bg behind, mid1/2/3 in creation order, fg in front with transparency
SAVE test6_complex_layering.png
EOF
