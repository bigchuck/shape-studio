# ========================================
@complex_layering
# TEST 7: Complex Layering
# ========================================
CLEAR WIP ALL
RESET_ZORDER

POLY bg 0,0 768,0 768,768 0,768
FILL bg lightgray
ZORDER bg -10

PROC dynamic_polygon mid1 VERTICES=5 BOUNDS=100,100,300,300
PROC dynamic_polygon mid2 VERTICES=6 BOUNDS=400,100,600,300
PROC dynamic_polygon mid3 VERTICES=7 BOUNDS=100,400,300,600

COLOR mid1 blue
COLOR mid2 green  
COLOR mid3 red

POLY fg 200,200 568,200 568,568 200,568
ALPHA fg 0.5
FILL fg yellow
ZORDER fg 100

INFO bg
INFO mid1
INFO mid2
INFO mid3
INFO fg

SAVE test7_complex_layering.png
EOF
