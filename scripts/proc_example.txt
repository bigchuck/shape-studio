# Example: Using PROC for dynamic polygon generation
# Demonstrates procedural generation with random parameters

# Clear the canvas
CLEAR WIP ALL

# Create a simple dynamic polygon with defaults
PROC dynamic_polygon shape1 VERTICES=5 BOUNDS=100,100,300,300

# Create one with a range (method will choose randomly)
PROC dynamic_polygon shape2 VERTICES=6,10 BOUNDS=400,100,700,300

# Use a preset configuration
PROC dynamic_polygon shape3 PRESET=organic BOUNDS=100,400,300,700

# Complex configuration with all parameters
PROC dynamic_polygon shape4 VERTICES=RAND(5,8) BOUNDS=400,400,700,700 ITERATIONS=20 DEPTH_RANGE=0.3,0.9 OPERATIONS=split_offset,sawtooth,squarewave

# Promote all to MAIN
PROMOTE shape1
PROMOTE shape2
PROMOTE shape3
PROMOTE shape4

# Clean up
CLEAR WIP ALL
